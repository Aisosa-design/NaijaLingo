import 'package:flutter/material.dart';

class HomeScreen extends StatelessWidget {
  final languages = ["Yoruba", "Igbo", "Hausa", "Tiv"];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('NaijaLingo')),
      body: ListView(
        children: languages.map((lang) {
          return ListTile(
            title: Text(lang),
            trailing: Icon(Icons.arrow_forward_ios),
            onTap: () => Navigator.pushNamed(context, '/lessons', arguments: lang),
          );
        }).toList(),
      ),
    );
  }
}
