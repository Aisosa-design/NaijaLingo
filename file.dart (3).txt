import 'package:http/http.dart' as http;
import 'dart:convert';

class ApiService {
  final baseUrl = '[localhost](http://localhost:4000)';

  Future<List<dynamic>> fetchLessons(String lang) async {
    final response = await http.get(Uri.parse('$baseUrl/lessons/$lang'));
    if (response.statusCode == 200) {
      return json.decode(response.body);
    } else {
      throw Exception('Failed to load lessons');
    }
  }
}
